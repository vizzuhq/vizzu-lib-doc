chart.animate({
    data: Object.assign(data, {
        filter: record =>
            ['AT', 'BE', 'DE', 'DK', 'ES', 'FI', 'FR', 'IT', 'NL', 'SE']
            .includes(record.Country_code)
    }),
    config: {
        channels: {
            x: { set: ['Year', 'Joy factors'] },
            y: { set: ['Value 3 (+)'] },
            color: { set: ['Country_code'] }
        },
        title: 'Stacked Area',
        geometry: 'line'
    }
});

    data: {
